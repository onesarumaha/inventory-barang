Struktur Dasar Template

Ini adalah struktur dasar template Reza Admin, selebihnya kamu bisa menambahkan sendiri, sesuai dokumentasi.

CopyHTML
<!DOCTYPE html>
<html lang="en">
<head>
    <!-- Meta tag wajib -->
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1, shrink-to-fit=no">

    <title>Dashboard . Reza Admin</title>

    <!-- Pertama Bootstrap CSS -->
    <link rel="stylesheet" href="[host mu]/dist/css/bootstrap.min.css">
    <!-- Kemudian Font Awesome -->
    <link rel="stylesheet" type="text/css" href="[host mu]/plugins/font-awesome-4.7.0/css/font-awesome.min.css">
    <!-- dan Reza Admin CSS -->
    <link rel="stylesheet" type="text/css" href="[host mu]/dist/css/reza-admin.min.css">
    <!-- Favicon -->
    <link rel="icon" href="[host mu]/dist/img/Reza_Admin.ico">
</head>
<body>

    <h1>Hello World</h1>

    <!-- Pertama jQuery, kemudian Bootstrap JS, dan Reza Admin JS -->
    <script src="[host mu]/dist/js/jquery-3.5.1.slim.min.js"></script>
    <script src="[host mu]/dist/js/bootstrap.min.js"></script>
    <script src="[host mu]/dist/js/reza-admin.min.js"></script>
</body>
</html>
